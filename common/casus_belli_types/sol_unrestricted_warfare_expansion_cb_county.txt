sol_expansion_war_county = {
	icon = county_crown
	group = sol_expansion
	ai = no

	cb_name = "SOL_EXPANSION_COUNTY_CB_NAME"
	war_name = "SOL_EXPANSION_COUNTY_WAR_NAME"
	war_name_base = "SOL_EXPANSION_COUNTY_WAR_NAME_BASE"

	target_de_jure_regions_above = yes
	use_de_jure_wargoal_only = yes

	target_titles = all
	target_title_tier = county
	target_de_jure_regions_above = yes
	target_top_liege_if_outside_realm = no
	
	combine_into_one = yes
	should_show_war_goal_subview = yes

	on_primary_attacker_death = inherit
	on_primary_defender_death = inherit
	attacker_allies_inherit = yes
	defender_allies_inherit = yes

	mutually_exclusive_titles = {
		always = no
	}

	#ignore_effect = change_title_holder

	allow_hostages = no
	white_peace_possible = no

	allowed_for_character = {
		is_ai = no

		NOT = {
			government_has_flag = government_is_landless_adventurer
		}
	}

	allowed_for_character_display_regardless = {
		is_ai = no
	}
	
	allowed_against_character = {
		scope:defender = {
			is_ai = yes
		}
	}

	valid_to_start = {
		scope:attacker = {
			is_ai = no
		}
	}

	should_invalidate = {
		NOT = {
			any_in_list = {
				list = target_titles
				any_in_de_jure_hierarchy = {
					tier = tier_county
					holder = {
						target_is_same_character_or_above = scope:defender
					}
				}
			}
		}
	}

	on_invalidated_desc = msg_expansion_war_invalidated_message
	
	on_invalidated = {

	}

	cost = {
		#gold = 1
		#piety = 1
		#prestige = 1
	}
	
	on_declaration = {
		on_declared_war = yes
		scope:attacker = {
			#add_gold = 100
			#add_piety = 50
			#add_prestige = 50

			spawn_army = { # Fighting Force 1
				name = sol_expanded_warfare_fighting_troops
				location = scope:attacker.capital_province
				origin = scope:attacker.capital_province
				uses_supply = no
				inheritable = no
				war = scope:war
				war_keep_on_attacker_victory = no

				levies = 800

				men_at_arms = { # 100
					type = accolade_maa_skirmishers
					stacks = 2
				}
				men_at_arms = { # 100
					type = accolade_maa_archers
					stacks = 2
				}
				men_at_arms = { # 100
					type = accolade_maa_horse_archers
					stacks = 2
				}
				men_at_arms = { # 100
					type = accolade_maa_outriders
					stacks = 2
				}
				men_at_arms = { # 50
					type = accolade_maa_vanguards
					stacks = 4
				}

			}


			every_realm_province = {
				if = {
					limit = {
						has_holding = yes
					}
					refill_levy = yes
					refill_garrison = yes
				}

				if = {
					limit = {
						has_province_modifier = recently_looted_modifier
					}
					remove_province_modifier = recently_looted_modifier
				}

			}

			every_maa_regiment = {
				limit = { maa_current_troops_count < maa_max_troops_count }
				change_maa_troops_count = replenishable_troops_value
			}

		}
	}

	on_victory_desc = {
		first_valid = {
			triggered_desc = {
				trigger = { scope:attacker = { is_local_player = yes } }
				desc = sol_expansion_war_victory_desc_attacker
			}
			desc = sol_expansion_war_victory_desc
		}

	}
	
	on_victory = {
		scope:attacker = { show_pow_release_message_effect = yes }
		
		# Legitimacy
		add_legitimacy_attacker_victory_effect = yes

		# Influence
		add_influence_attacker_victory_effect = yes

		#EP2 accolade glory gain for winning against higher ranked enemy
		scope:attacker = { accolade_attacker_war_end_glory_gain_high_effect = yes }

		scope:attacker = {
			add_gold = 100
			add_piety = 100
			add_prestige = 100
			add_dynasty_prestige = 100

			every_realm_county = {
				change_county_control = 50
			}

			every_realm_province = {


				if = {
					limit = {
						has_holding = yes
					}
					refill_levy = yes
					refill_garrison = yes
				}

				if = {
					limit = {
						has_province_modifier = recently_looted_modifier
					}
					remove_province_modifier = recently_looted_modifier
				}

			}

			every_maa_regiment = {
				limit = { maa_current_troops_count < maa_max_troops_count }
				change_maa_troops_count = replenishable_troops_value
			}

		}




#####	IMPRISON THE LOSERS
		scope:attacker = {	#Grab suitable prisoners from the defender.
			custom_tooltip = sol_captive_war_cb_prisoners.tt_attacker
			hidden_effect = {
				scope:defender = {
					every_courtier_or_guest = { add_to_list = potential_captives_list }
					every_close_or_extended_family_member = { add_to_list = potential_captives_list }
					every_vassal = { add_to_list = potential_captives_list }
					every_in_list = {	#Nab someone related to the defender, preferring tangentials.
						list = potential_captives_list
						limit = {
							OR = {
								AND = {
									has_dynasty = yes
									dynasty = scope:defender.dynasty
								}
								#is_close_family_of = scope:defender
								is_consort_of = scope:defender
								is_spouse_of = scope:defender
								#is_extended_family_of = scope:defender
							}
						}

						add_to_list = definite_captives_list
						save_temporary_scope_as = future_captive
						scope:attacker = {
							imprison = {
								target = scope:future_captive
								type = dungeon
							}
						}
					}

				}

			}
		}

		create_title_and_vassal_change = {
			type = conquest
			save_scope_as = change
			add_claim_on_loss = no
		}

		show_as_tooltip = {
			every_in_list = {
				list = target_titles
				custom_tooltip = expansion_title_transfer_tt
				custom_tooltip = expansion_title_transfer_2_tt
				every_in_de_jure_hierarchy = {
					limit = { holder.top_liege = scope:defender }
					change_title_holder = {
						holder = scope:attacker
						change = scope:change
						take_baronies = yes
					}
				}
			}
			scope:defender = {
				every_sub_realm_county = {
					limit = {
						county_controller = scope:attacker
					}
					change_title_holder = {
						holder = scope:attacker
						change = scope:change
						take_baronies = yes
					}
				}
			}
		}
		
		hidden_effect = {
			setup_invasion_cb = {
				titles = target_titles
				attacker = scope:attacker
				defender = scope:defender
				change = scope:change
				take_occupied = yes
				}
			resolve_title_and_vassal_change = scope:change
		}

		# Attacker gets Prestige Experience, Defender loses Prestige, all other participants gain Prestige based on their war contribution.
		modify_all_participants_fame_values = {
			WINNER = scope:attacker
			LOSER = scope:defender
			FAME_BASE = scope:cb_prestige_factor # Set by 'setup_invasion_cb'
			IS_RELIGIOUS_WAR = no
			WINNER_FAME_SCALE = 10
			LOSER_FAME_SCALE = -10
			WINNER_ALLY_FAME_SCALE = 10
			LOSER_ALLY_FAME_SCALE = 10
		}

		# Truce
		#add_truce_attacker_victory_effect = yes

		# FP1: note the victory for future memorialisation via stele (if applicable).
		scope:attacker = { fp1_remember_recent_conquest_victory_effect = yes }

		# EP3: note gold gained from military assistance/join war contracts and their war contribution threshold
		laamp_as_mercenary_payout_tooltip_effect = yes
	}

	on_white_peace_desc = {
		first_valid = {
			triggered_desc = {
				trigger = { scope:defender = { is_local_player = yes } }
				desc = sol_expansion_war_white_peace_desc_defender
			}
			desc = sol_expansion_war_white_peace_desc
		}
	}
	

	on_defeat_desc = {
		first_valid = {
			triggered_desc = {
				trigger = { scope:defender = { is_local_player = yes } }
				desc = sol_expansion_war_white_peace_desc_defender
			}
			desc = sol_expansion_war_white_peace_desc
		}
	}

	on_defeat = {
		scope:attacker = { show_pow_release_message_effect = yes }
		
		# Legitimacy
		add_legitimacy_attacker_defeat_effect = yes

		# Influence
		add_influence_attacker_defeat_effect = yes

		#EP2 accolade glory gain for winning against higher ranked enemy
		scope:defender = { accolade_defender_war_end_glory_gain_high_effect = yes }

		# Prestige loss for the attacker
		scope:attacker = {
			pay_short_term_gold_reparations_effect = {
				GOLD_VALUE = 5
			}
			add_prestige = {
				value = massive_prestige_value
				multiply = -1.0
			}
		}
		
		setup_invasion_cb = {
			titles = target_titles
			attacker = scope:attacker
			defender = scope:defender
			claimant = scope:claimant
			victory = no
		}

		# Attacker loses Prestige, all other war participants gain Prestige (Defender gains full prestige, all allies on both sides gain based on war contribution).
		modify_all_participants_fame_values = {
			WINNER = scope:defender
			LOSER = scope:attacker
			FAME_BASE = scope:cb_prestige_factor # Set by 'setup_claim_cb'
			IS_RELIGIOUS_WAR = no
			WINNER_FAME_SCALE = scale_10_war_defender_win
			LOSER_FAME_SCALE = -10
			WINNER_ALLY_FAME_SCALE = 10
			LOSER_ALLY_FAME_SCALE = 10
		}
	
		#add_truce_attacker_defeat_effect = yes
		
		scope:attacker = {
			save_temporary_scope_as = loser
		}
		on_lost_aggression_war_discontent_loss = yes

		# EP3: note gold gained from military assistance/join war contracts and their war contribution threshold
		laamp_as_mercenary_payout_tooltip_effect = yes
	}
	
	transfer_behavior = transfer


		
	interface_priority = 100
	attacker_wargoal_percentage = 0.8
	
	max_defender_score_from_occupation = 150
	max_attacker_score_from_occupation = 150
}
