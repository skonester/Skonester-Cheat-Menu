special_gift_interaction_2 = {
	category = interaction_category_uncategorized
	interface_priority = 60
	desc = special_gift_interaction_desc_2

	ai_frequency = 0

	is_shown = {
		scope:actor = {
			has_trait = water_of_life
		}
		scope:recipient = {
			NOT = {
				is_adult = no
				has_trait = water_of_life
			}	
		}
		scope:recipient = {
			is_spouse_of = scope:actor
		}
	}

	on_accept = {
		scope:actor = {
			send_interface_message = {
				type = event_generic_neutral
				title = "Gift of Immortality"
				custom_tooltip = special_gift_interaction_tooltip_2
				right_icon = scope:recipient
				
				scope:recipient = {
					target = scope:actor
					add_trait = water_of_life
				}
			}
		}
	}

	auto_accept = yes

	ai_min_reply_days = 1
	ai_max_reply_days = 1
	
	ai_will_do = {
		base = 0
	}
}
