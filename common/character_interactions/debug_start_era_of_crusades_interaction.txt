dp_debug_start_era_of_crusades_interaction = {
    category = interaction_pcm_main
    icon = on
    use_diplomatic_range = no
    ignores_pending_interaction_block = yes

    is_shown = {
        scope:actor = { 
            is_ai = no 
            has_character_flag = bone_cheat_interactions_flag
        }
        faith:catholic = {
            NOT = {
                religion = { exists = var:variable_ghw_unlocked }
            }
        }
    }

    auto_accept = yes

    on_accept = {
        faith:catholic = {
            change_fervor = 200
            save_scope_as = awakening_faith

            if = {
                limit = { exists = religious_head }
                religious_head = { save_scope_as = ghw_first_sponsor }
            }
            else = {
                random_theocratic_ruler = {
                    limit = { faith = root }
                    save_scope_as = ghw_first_sponsor
                }
            }

            if = {
                limit = { NOT = { exists = scope:ghw_first_sponsor } }
                random_ruler = {
                    limit = { faith = root }
                    save_scope_as = ghw_first_sponsor
                }
            }

            religion = { set_variable = { name = variable_ghw_unlocked value = yes } }

            every_player = {
                trigger_event = great_holy_war.0011
            }
        }
    }
}

dp_debug_launch_natural_crusade_interaction = {
    category = interaction_pcm_main
    icon = on
    use_diplomatic_range = no
    ignores_pending_interaction_block = yes
    is_shown = {
        scope:actor = { 
            is_ai = no
            has_character_flag = bone_cheat_interactions_flag
        }
        OR = {
            faith:catholic = {
                NOR = {
                    exists = great_holy_war
                    exists = var:variable_ghw_cooldown
                    religion = { exists = var:variable_first_ghw_cooldown }
                }
                religion = { exists = var:variable_ghw_unlocked }
            }
            faith:orthodox = {
                NOR = {
                    exists = great_holy_war
                    exists = var:variable_ghw_cooldown
                    religion = { exists = var:variable_first_ghw_cooldown }
                }
                religion = { exists = var:variable_ghw_unlocked }
            }
            faith:apostolic = {
                NOR = {
                    exists = great_holy_war
                    exists = var:variable_ghw_cooldown
                    religion = { exists = var:variable_first_ghw_cooldown }
                }
                religion = { exists = var:variable_ghw_unlocked }
            }
        }
    }
    auto_accept = yes
    on_accept = {
        OR = {
            faith:catholic = { change_fervor = 200 }
            faith:orthodox = { change_fervor = 200 }
            faith:apostolic = { change_fervor = 200 }
        }
        trigger_event = great_holy_war.0001
    }
}

dp_debug_start_great_holy_war_interaction = {
    category = interaction_pcm_main
    icon = on
    use_diplomatic_range = no
    ignores_pending_interaction_block = yes
    is_shown = {
        scope:actor = { 
            is_ai = no
            has_character_flag = bone_cheat_interactions_flag
        }
        NOT = { scope:recipient.faith = scope:actor.faith }
    }
    auto_accept = yes
    on_accept = {
        scope:actor.faith = {
            start_great_holy_war = {
                target_character = scope:recipient
                target_title = scope:recipient.primary_title
                delay = 110
            }
        }
    }
}
