pcm_convert_subrealm_to_culture_decision = {
	decision_group_type = pcm
	picture = { reference = gfx/interface/illustrations/decisions/pcm_decision.dds }
	title = pcm_subrealm_to_culture_decision_title
    desc = pcm_subrealm_to_culture_decision_desc
    selection_tooltip = pcm_subrealm_to_culture_decision_tooltip
    confirm_text = pcm_subrealm_to_culture_decision_ct

	is_shown = {
		is_ai = no
		has_character_flag = pcm_show_decision_flag
	}

	effect = {
		every_sub_realm_county = {
			limit = { NOT = { culture = root.culture } }
			set_county_culture = root.culture
		}
		every_vassal_or_below = {
			limit = { NOT = { culture = root.culture } }
			set_culture = root.culture
		}
		every_child = {
			limit = { NOT = { culture = root.culture} }
			set_culture = root.culture
		}
		every_consort = {
			limit = { NOT = { culture = root.culture } }
			set_culture = root.culture
		}
		every_close_family_member = {
			limit = { NOT = { culture = root.culture } }
			set_culture = root.culture
		}
		every_heir = {
			limit = { NOT = { culture = root.culture } }
			set_culture = root.culture
		}
		house = {
			every_house_member = {
				limit = {
					is_alive = yes
					NOT = { culture = root.culture }
				}
				set_culture = root.culture
			}
		}
	}
    ai_will_do = { base = 0 }
	ai_check_interval = 0
}
