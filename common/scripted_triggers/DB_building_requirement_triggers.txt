building_caravan_requirement = {
	AND = {
		OR = {
			terrain = oasis
			terrain = steppe
			terrain = drylands
			terrain = desert
			terrain = desert_mountains
			terrain = plains
			terrain = mountains
			terrain = hills
		}
		NOT = { geographical_region = world_europe }
	}
}

				has_cultural_pillar = heritage_goidelic
			}
		}
	}
}
