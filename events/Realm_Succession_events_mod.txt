change_Realm_succession_law_mod.1001 = {
    type = character_event
    theme = default
	desc = change_Realm_succession_law_mod.1001.desc

	trigger = {
		is_ai = no # Only players should trigger the main menu
		is_landless_adventurer = no
		NOT = { has_government = administrative_government }
	}

	option = {
		#Multiple Heirs
		name = multiple_heirs_rse_mod
		trigger_event = change_Realm_succession_law_mod.1002
		trigger = {
			or = {
				has_game_rule = change_Realm_succession_law_game_mod_on
				culture = {
					or = {
						has_innovation = innovation_heraldry
						has_innovation = innovation_hereditary_rule
						has_innovation = innovation_visigothic_codes
						has_innovation = innovation_gavelkind
					}
				}
			}
		}
	}
	option = {
		#Single Heir
		name = single_Heir_rse_mod
		trigger_event = change_Realm_succession_law_mod.1006
		trigger = {
			or = {
				has_game_rule = change_Realm_succession_law_game_mod_on
				culture = {
					or = {
						has_cultural_pillar = heritage_west_slavic
						has_innovation = innovation_primogeniture
						has_innovation = innovation_heraldry
						has_innovation = innovation_table_of_princes
						has_innovation = innovation_theme_system
					}
				}
			}
		}
	}
	option = {
		name = "Unlock Innovation"
		trigger_event = change_Realm_succession_law_mod.1010
		trigger = {
			nand = {
				root.culture = {
					has_innovation = innovation_hereditary_rule
				}
				root.culture = {
					has_innovation = innovation_heraldry
				}
				root.culture = {
					has_innovation = innovation_primogeniture
				}
				root.culture = {
					has_innovation = innovation_theme_system
				}
			}
		}
	}
	# option = {#Elective Succession
	# name = Elective_rse_mod
	# trigger_event = change_Realm_succession_law_mod.1010}
}

change_Realm_succession_law_mod.1003 = {
    type = character_event
    theme = default
	#Confederate Partition
	desc = change_Realm_succession_law_mod.1003.desc
	option = {
		name = Only_Me_gse_mod
		add_realm_law_skip_effects = confederate_partition_succession_law
		add_character_flag = rse_flag_confederate_partition
	}
	option = {
		name = Me_and_Vassals_gse_mod
		add_realm_law_skip_effects = confederate_partition_succession_law
		every_vassal_or_below = {
			limit = {
				or = {
					has_government = clan_government
					has_government = feudal_government
					has_government = tribal_government
				}
				is_landless_adventurer = no
			}
			add_realm_law_skip_effects = confederate_partition_succession_law
			remove_character_flag = rse_flag_confederate_partition
			remove_character_flag = rse_flag_partition
			remove_character_flag = rse_flag_high_partition
			remove_character_flag = rse_flag_primogeniture
			remove_character_flag = rse_flag_ultimogeniture
			remove_character_flag = rse_flag_house_seniority
			add_character_flag = rse_flag_confederate_partition
		}
	}
	option = {
		name = Only_Vassals_gse_mod
		every_vassal_or_below = {
			limit = {
				or = {
					has_government = clan_government
					has_government = feudal_government
					has_government = tribal_government
				}
				is_landless_adventurer = no
			}
			add_realm_law_skip_effects = confederate_partition_succession_law
			remove_character_flag = rse_flag_confederate_partition
			remove_character_flag = rse_flag_partition
			remove_character_flag = rse_flag_high_partition
			remove_character_flag = rse_flag_primogeniture
			remove_character_flag = rse_flag_ultimogeniture
			remove_character_flag = rse_flag_house_seniority
			add_character_flag = rse_flag_confederate_partition
		}
	}
	option = {
		name = Nothing_will_change_gse_mod
	}
}

change_Realm_succession_law_mod.1010 = {
    type = character_event
    theme = default
	title = "Innovation"
	trigger = {
		character = ROOT
	}
	option = {
		name = "Hereditary Rule"
		trigger = {
			not = {
				root.culture = {
					has_innovation = innovation_hereditary_rule
				}
			}
		}
		root.culture = {
			add_innovation = innovation_hereditary_rule
		}
	}
	option = {
		name = "Heraldry"
		trigger = {
			not = {
				root.culture = {
					has_innovation = innovation_heraldry
				}
			}
		}
		root.culture = {
			add_innovation = innovation_heraldry
		}
	}
	option = {
		name = "Primogeniture"
		trigger = {
			not = {
				root.culture = {
					has_innovation = innovation_primogeniture 
				}
			}
		}
		root.culture = {
			add_innovation = innovation_primogeniture 
		}
	}
	option = {
		name = "Theme System"
		trigger = {
			not = {
				root.culture = {
					has_innovation = innovation_theme_system
				}
			}
		}
		root.culture = {
			add_innovation = innovation_theme_system
		}
	}
	option = {
		name = "Close"
	}
}
