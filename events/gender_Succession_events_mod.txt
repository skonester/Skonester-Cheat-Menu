namespace = gender_succession_events_mod

gender_succession_events_mod.1001 = {
    type = character_event
    theme = default
	desc = gender_succession_events_mod.1001.desc

	trigger = {
		is_ai = no
		is_landless_adventurer = no
		NOT = { has_government = administrative_government }
	}

	#Male Only
	option = {
		name = male_only_gse_mod
		trigger_event = gender_succession_events_mod.1002
	}
	#Male preference
	option = {
		name = Male_preference_gse_mod
		trigger_event = gender_succession_events_mod.1003
	}
	#Equal
	option = {
		name = Equal_gse_mod
		trigger_event = gender_succession_events_mod.1004
	}
	#Female preference

	}
	#Female Only

	}
}

gender_succession_events_mod.1002 = {
    type = character_event
    theme = default
	#Male Only
	desc = gender_succession_events_mod.1002.desc
	option = {
		name = Only_Me_gse_mod
		add_realm_law_skip_effects = male_only_law
		gender_rse_flag_cleanup = yes
		add_character_flag = rse_flag_male_only
	}
	option = {
		name = Me_and_Vassals_gse_mod
		add_realm_law_skip_effects = male_only_law
		gender_rse_flag_cleanup = yes
		add_character_flag = rse_flag_male_only
		every_vassal_or_below = {
			limit = {
				is_landless_adventurer = no
				NOT = { has_government = administrative_government }
			}
			add_realm_law_skip_effects = male_only_law
			gender_rse_flag_cleanup = yes
			add_character_flag = rse_flag_male_only
		}
	}
	option = {
		name = Only_Vassals_gse_mod
		every_vassal_or_below = {
			limit = {
				is_landless_adventurer = no
				NOT = { has_government = administrative_government }
			}
			add_realm_law_skip_effects = male_only_law
			gender_rse_flag_cleanup = yes
			add_character_flag = rse_flag_male_only
		}
	}
	option = {
		name = Nothing_will_change_gse_mod
	}
}
