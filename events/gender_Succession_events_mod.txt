gender_succession_events_mod.1002 = {
    type = character_event
    theme = default
	#Male Only
	desc = gender_succession_events_mod.1002.desc
	option = {
		name = Only_Me_gse_mod
		add_realm_law_skip_effects = male_only_law
	}
	option = {
		name = Me_and_Vassals_gse_mod
		add_realm_law_skip_effects = male_only_law
		every_vassal_or_below = {
			limit = {
				or = {
					has_government = clan_government
					has_government = feudal_government
					has_government = tribal_government
				}
				nor = {
					has_government = republic_government
					has_government = theocracy_government
					has_government = mercenary_government
					has_government = holy_order_government
				}
			}
			add_realm_law_skip_effects = male_only_law
		}
	}
	option = {
		name = Only_Vassals_gse_mod
		every_vassal_or_below = {
			limit = {
				or = {
					has_government = clan_government
					has_government = feudal_government
					has_government = tribal_government
				}
				nor = {
					has_government = republic_government
					has_government = theocracy_government
					has_government = mercenary_government
					has_government = holy_order_government
				}
			}
			add_realm_law_skip_effects = male_only_law
		}
	}
	option = {
		name = Nothing_will_change_gse_mod
	}
}
